    """–ù–æ—Ä–º–∞–ª–∏–∑—É–µ—Ç –≠–ú–ì-–¥–∞–Ω–Ω—ã–µ —Å —Ä–∞–∑–Ω—ã—Ö –∞–ø–ø–∞—Ä–∞—Ç–æ–≤ –∫ –µ–¥–∏–Ω–æ–π —à–∫–∞–ª–µ"""
    
    def normalize_to_control(self, value: float, apparatus: EMGApparatus, 
                           condition: MeasurementCondition, 
                           muscle: MuscleType) -> float:
        """
        –ù–æ—Ä–º–∞–ª–∏–∑—É–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –∫ –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω—ã–º –ø–æ–∫–∞–∑–∞—Ç–µ–ª—è–º Synapsys

        1. –ü–æ–ª—É—á–∞–µ–º –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –¥–∞–Ω–Ω–æ–≥–æ –∞–ø–ø–∞—Ä–∞—Ç–∞
        2. –ü–æ–ª—É—á–∞–µ–º –∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è Synapsys (—ç—Ç–∞–ª–æ–Ω)

**–ü—Ä–∏—á–∏–Ω–∞:** –ù–µ–∫–æ—Ç–æ—Ä—ã–µ –∫–ª–∞—Å—Å—ã –∫–æ–º–ø–æ–∑–∏—Ç–æ–≤ –∏–º–µ–ª–∏ —Ç–æ–ª—å–∫–æ 1 –ø—Ä–∏–º–µ—Ä.

**–†–µ—à–µ–Ω–∏–µ:** –î–æ–±–∞–≤–ª–µ–Ω–∞ –≥–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∏–Ω—Ç–µ—Ç–∏—á–µ—Å–∫–∏—Ö –ø—Ä–∏–º–µ—Ä–æ–≤ –¥–ª—è –∫–ª–∞—Å—Å–æ–≤ —Å –º–∞–ª—ã–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º –¥–∞–Ω–Ω—ã—Ö:

```python
def augment_data(X, y):
    """–î–æ–±–∞–≤–ª—è–µ—Ç —Å–∏–Ω—Ç–µ—Ç–∏—á–µ—Å–∫–∏–µ –ø—Ä–∏–º–µ—Ä—ã –¥–ª—è –∫–ª–∞—Å—Å–æ–≤ —Å –º–∞–ª—ã–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ–º –¥–∞–Ω–Ω—ã—Ö"""
    from collections import Counter
    class_counts = Counter(y)
    min_samples = 3  # –ú–∏–Ω–∏–º—É–º –ø—Ä–∏–º–µ—Ä–æ–≤ –Ω–∞ –∫–ª–∞—Å—Å
    
    augmented_X = list(X)
    augmented_y = list(y)
    
    for class_name, count in class_counts.items():
        if count < min_samples:
            # –ù–∞—Ö–æ–¥–∏–º –ø—Ä–∏–º–µ—Ä—ã —ç—Ç–æ–≥–æ –∫–ª–∞—Å—Å–∞
            class_indices = [i for i, label in enumerate(y) if label == class_name]
            
            # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–Ω—Ç–µ—Ç–∏—á–µ—Å–∫–∏–µ –ø—Ä–∏–º–µ—Ä—ã
            for _ in range(min_samples - count):
                base_idx = np.random.choice(class_indices)
                # –î–æ–±–∞–≤–ª—è–µ–º –Ω–µ–±–æ–ª—å—à–æ–π —à—É–º
                synthetic = X[base_idx] + np.random.normal(0, 0.1, size=X[base_idx].shape)
                augmented_X.append(synthetic)
                augmented_y.append(class_name)
    
    return np.array(augmented_X), np.array(augmented_y)
```
